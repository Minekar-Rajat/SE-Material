
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity structure_ha is
    Port ( a : in  STD_LOGIC;
           b : in  STD_LOGIC;
           sum : out  STD_LOGIC;
           cout : out  STD_LOGIC);
end structure_ha;

architecture structure_HA of structure_ha is

begin
sum<= a xor b;
cout<= a and b;

end structure_HA;
